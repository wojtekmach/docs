1use std::collections::HashMap;

3use std::net::{TcpListener, TcpStream};
4use std::sync::{mpsc, Arc, Mutex, OnceLock};

7type Callback = Box<dyn Fn(&[u8]) + Send>;
8
9static STATE: OnceLock<PubSubState> = OnceLock::new();
10static BIND_LOCK: Mutex<()> = Mutex::new(());
11
12struct PubSubState {
13    stream: Arc<Mutex<Option<TcpStream>>>,
14    subscribe_tx: mpsc::Sender<(String, Callback)>,
15}
16
17/// A handle to the PubSub connection.
18#[derive(Clone, Copy)]
19pub struct PubSub {
20    port: u16,
21}
22
23impl PubSub {
24    /// Binds a TCP listener and returns a [`PubSub`] handle.
25    pub fn bind(tcp_port: u16) -> Result<Self, io::Error> {
26        let _guard = BIND_LOCK.lock().unwrap();
27
28        if STATE.get().is_some() {
29            return Err(io::Error::new(
30                io::ErrorKind::AlreadyExists,
31                "pubsub already bound",
32            ));
33        }
34
35        let listener = TcpListener::bind(("127.0.0.1", tcp_port))?;
36        let actual_port = listener.local_addr()?.port();
37
38        let stream: Arc<Mutex<Option<TcpStream>>> = Arc::new(Mutex::new(None));
39        let (subscribe_tx, subscribe_rx) = mpsc::channel::<(String, Callback)>();
40
41        let stream_ref = stream.clone();
42        thread::Builder::new()
43            .name("elixirkit-pubsub".into())
44            .spawn(move || {
45                let Ok((tcp_stream, _)) = listener.accept() else {
46                    return;
47                };
48                let _ = tcp_stream.set_nodelay(true);
49
50                let reader = match tcp_stream.try_clone() {
51                    Ok(r) => r,
52                    Err(_) => return,
53                };
54                *stream_ref.lock().unwrap() = Some(tcp_stream);
55
56                read_loop(subscribe_rx, reader);
57                *stream_ref.lock().unwrap() = None;
58            })?;
59
60        assert!(
61            STATE
62                .set(PubSubState {
63                    stream,
64                    subscribe_tx,
65                })
66                .is_ok(),
67            "bind lock guarantees single initialization"
68        );
69
70        Ok(PubSub { port: actual_port })
71    }
72
73    /// Returns the port number the listener is bound to.
74    pub fn port(&self) -> u16 {
75        self.port
76    }
77
78    /// Subscribes to messages on the given topic from the Elixir side.
79    pub fn subscribe<F>(&self, topic: &str, callback: F)
80    where
81        F: Fn(&[u8]) + Send + 'static,
82    {
83        let _ = state()
84            .subscribe_tx
85            .send((topic.to_string(), Box::new(callback)));
86    }
87
88    /// Broadcasts a message on the given topic to the Elixir side.
89    pub fn broadcast(&self, topic: &str, message: &[u8]) -> io::Result<()> {
90        let guard = state().stream.lock().unwrap();
91        match guard.as_ref() {
92            Some(stream) => write_message(stream, topic.as_bytes(), message),
93            None => Err(io::Error::new(io::ErrorKind::NotConnected, "not connected")),
94        }
95    }
96}
97
98fn state() -> &'static PubSubState {
99    STATE.get().expect("pubsub not bound")
100}

102fn read_loop(subscribe_rx: mpsc::Receiver<(String, Callback)>, mut reader: TcpStream) {
103    let mut subscribers: HashMap<String, Vec<Callback>> = HashMap::new();
104
105    loop {
106        while let Ok((topic, callback)) = subscribe_rx.try_recv() {
107            subscribers.entry(topic).or_default().push(callback);
108        }
109
110        let mut len_buf = [0u8; 4];
111        if reader.read_exact(&mut len_buf).is_err() {
112            break;
113        }
114        let frame_len = u32::from_be_bytes(len_buf) as usize;
115
116        let mut frame = vec![0u8; frame_len];
117        if reader.read_exact(&mut frame).is_err() {
118            break;
119        }
120
121        if frame.is_empty() {
122            continue;
123        }
124        let topic_len = frame[0] as usize;
125        if frame.len() < 1 + topic_len + 4 {
126            continue;
127        }
128        let topic = &frame[1..1 + topic_len];
129        let payload_len_offset = 1 + topic_len;
130        let payload_len = u32::from_be_bytes([
131            frame[payload_len_offset],
132            frame[payload_len_offset + 1],
133            frame[payload_len_offset + 2],
134            frame[payload_len_offset + 3],
135        ]) as usize;
136        let payload_offset = payload_len_offset + 4;
137        if frame.len() < payload_offset + payload_len {
138            continue;
139        }
140        let payload = &frame[payload_offset..payload_offset + payload_len];
141
142        let topic_str = String::from_utf8_lossy(topic);
143
144        if let Some(callbacks) = subscribers.get(topic_str.as_ref()) {
145            for cb in callbacks {
146                cb(payload);
147            }
148        }
149    }
150}

152fn write_message(mut stream: &TcpStream, topic: &[u8], payload: &[u8]) -> io::Result<()> {
153    let inner_len = 1 + topic.len() + 4 + payload.len();
154    let frame_len = u32::try_from(inner_len)
155        .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "frame too large"))?;
156    let payload_len = u32::try_from(payload.len())
157        .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "payload too large"))?;
158
159    stream.write_all(&frame_len.to_be_bytes())?;
160    stream.write_all(&[topic.len() as u8])?;
161    stream.write_all(topic)?;
162    stream.write_all(&payload_len.to_be_bytes())?;
163    stream.write_all(payload)?;
164    stream.flush()
165}

167/// Returns a command for running `elixir` with [`PubSub`].
168pub fn elixir(args: &[&str], pubsub: PubSub) -> std::process::Command {
169    let mut cmd = command(if cfg!(target_os = "windows") { "elixir.bat" } else { "elixir" }, pubsub);
170    cmd.args(args);
171    cmd
172}
173
174/// Returns a command for running a Mix task with [`PubSub`].
175pub fn mix(task: &str, args: &[&str], pubsub: PubSub) -> std::process::Command {
176    let mut cmd = command(if cfg!(target_os = "windows") { "mix.bat" } else { "mix" }, pubsub);
177    cmd.arg(task);
178    cmd.args(args);
179    cmd
180}

182/// Returns a command for launching an Elixir release with [`PubSub`].
183pub fn release(dir: impl AsRef<std::path::Path>, name: &str, pubsub: PubSub) -> std::process::Command {
184    let dir = dir.as_ref();
185    let mut script = dir.join("bin").join(name);
186    if cfg!(target_os = "windows") {
187        script.set_extension("bat");
188    }
189    let mut cmd = command(script.to_str().unwrap(), pubsub);
190    cmd.args(["start"]);
191    cmd
192}
193
194fn command(program: &str, pubsub: PubSub) -> std::process::Command {
195    let mut cmd = std::process::Command::new(program);
196    cmd.env("ELIXIRKIT_PORT", pubsub.port().to_string());
197    cmd
